Simple

$('#date1').MdPersianDateTimePicker({
    targetTextSelector: '#inputDate1',
    targetDateSelector: '#inputDate1-1',
    dateFormat: 'yyyy-MM-dd hh:mm:ss',
    isGregorian: true,
    enableTimePicker: true,
    disableBeforeToday: true,
    disabledDates: [new Date(),
        new Date(2018, 9, 11), 
        new Date(2018, 9, 12), 
        new Date(2018, 9, 13), 
        new Date(2018, 9, 14)],
    disabledDays: [5, 6]
});
